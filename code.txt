const carros = [
    {
        marca: 'Ford',
        modelo: 'Fiesta',
        passageiros: 4,
        road: ["Off-road", "Highway", "City"]
    },
    {
        marca: 'Citroen',
        modelo: 'C3',
        passageiros: 4,
        road: ["Highway", "City"]
    },
    {
        marca: 'Chevrolet',
        modelo: 'Onix',
        passageiros: 2,
        road: ["Highway", "City"],
    }
]

const passageiros = 3
const uso = "Off-road"

console.log("Primeiro filtro:")
